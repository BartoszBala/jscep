73  	/**
74  	 * Decodes the provided PKCS #7 <tt>signedData</tt> into a
75  	 * <tt>PkiMessage</tt>
76  	 * 
77  	 * @param pkiMessage
78  	 *            the <tt>signedData</tt> to decode.
79  	 * @return the decoded <tt>PkiMessage</tt>
80  	 * @throws MessageDecodingException
81  	 *             if there is a problem decoding the <tt>signedData</tt>
82  	 */
83  	@SuppressWarnings("unchecked")
84  	public PkiMessage<?> decode(CMSSignedData pkiMessage)
85  			throws MessageDecodingException {
86  		LOGGER.debug("Decoding pkiMessage");
87  		validate(pkiMessage);
88  
89  		// The signed content is always an octet string
90  		CMSProcessable signedContent = pkiMessage.getSignedContent();
91  
92  		SignerInformationStore signerStore = pkiMessage.getSignerInfos();
93  		SignerInformation signerInfo = signerStore.get(new JcaSignerId(signer));
94  		if (signerInfo == null) {
95  			throw new MessageDecodingException("Could not for signerInfo for "
96  					+ signer.getIssuerDN());
97  		}
98  
99  		LOGGER.debug("pkiMessage digest algorithm: {}", signerInfo
100 				.getDigestAlgorithmID().getAlgorithm());
101 		LOGGER.debug("pkiMessage encryption algorithm: {}",
102 				signerInfo.getEncryptionAlgOID());
103 
104 		Store store = pkiMessage.getCertificates();
105 		Collection<?> certColl;
106 		try {
107 			certColl = store.getMatches(signerInfo.getSID());
108 		} catch (StoreException e) {
109 			throw new MessageDecodingException(e);
110 		}
111 		if (certColl.size() > 0) {
112 			X509CertificateHolder cert = (X509CertificateHolder) certColl
113 					.iterator().next();
114 			LOGGER.debug(
115 					"Verifying pkiMessage using key belonging to [issuer={}; serial={}]",
116 					cert.getIssuer(), cert.getSerialNumber());
117 			SignerInformationVerifier verifier;
118 			try {
119 				verifier = new JcaSimpleSignerInfoVerifierBuilder().build(cert);
120 				signerInfo.verify(verifier);
121 
122 				LOGGER.debug("pkiMessage verified.");
123 			} catch (CMSException e) {
124 				throw new MessageDecodingException(e);
125 			} catch (OperatorCreationException e) {
126 				throw new MessageDecodingException(e);
127 			} catch (CertificateException e) {
128 				throw new MessageDecodingException(e);
129 			}
130 		} else {
131 			LOGGER.warn("Unable to verify message because the signedData contained no certificates.");
132 		}
133 
134 		Hashtable<DERObjectIdentifier, Attribute> attrTable = signerInfo
135 				.getSignedAttributes().toHashtable();
136 
137 		if (LOGGER.isDebugEnabled()) {
138 			LOGGER.debug("pkiMessage has {} signed attributes:", signerInfo
139 					.getSignedAttributes().size());
140 			for (Entry<DERObjectIdentifier, Attribute> entry : attrTable
141 					.entrySet()) {
142 				LOGGER.debug("  {}: {}", entry.getKey().getId(), entry
143 						.getValue().getAttrValues());
144 			}
145 		}
146 
147 		MessageType messageType = toMessageType(attrTable
148 				.get(toOid(MESSAGE_TYPE)));
149 		Nonce senderNonce = toNonce(attrTable.get(toOid(SENDER_NONCE)));
150 		TransactionId transId = toTransactionId(attrTable.get(toOid(TRANS_ID)));
151 
152 		if (messageType == MessageType.CERT_REP) {
153 			PkiStatus pkiStatus = toPkiStatus(attrTable.get(toOid(PKI_STATUS)));
154 			Nonce recipientNonce = toNonce(attrTable
155 					.get(toOid(RECIPIENT_NONCE)));
156 
157 			if (pkiStatus == PkiStatus.FAILURE) {
158 				FailInfo failInfo = toFailInfo(attrTable.get(toOid(FAIL_INFO)));
159 				LOGGER.debug("Finished decoding pkiMessage");
160 				return new CertRep(transId, senderNonce, recipientNonce,
161 						failInfo);
162 			} else if (pkiStatus == PkiStatus.PENDING) {
163 				LOGGER.debug("Finished decoding pkiMessage");
164 				return new CertRep(transId, senderNonce, recipientNonce);
165 			} else {
166 				final CMSEnvelopedData ed = getEnvelopedData(signedContent
167 						.getContent());
168 				final byte[] envelopedContent = decoder.decode(ed);
169 				CMSSignedData messageData;
170 				try {
171 					messageData = new CMSSignedData(envelopedContent);
172 				} catch (CMSException e) {
173 					throw new MessageDecodingException(e);
174 				}
175 				LOGGER.debug("Finished decoding pkiMessage");
176 				return new CertRep(transId, senderNonce, recipientNonce,
177 						messageData);
178 			}
179 		} else {
180 			CMSEnvelopedData ed = getEnvelopedData(signedContent.getContent());
181 			byte[] decoded = decoder.decode(ed);
182 			if (messageType == MessageType.GET_CERT) {
183 				IssuerAndSerialNumber messageData = IssuerAndSerialNumber
184 						.getInstance(decoded);
185 				LOGGER.debug("Finished decoding pkiMessage");
186 				return new GetCert(transId, senderNonce, messageData);
187 			} else if (messageType == MessageType.GET_CERT_INITIAL) {
188 				IssuerAndSubject messageData = new IssuerAndSubject(decoded);
189 				LOGGER.debug("Finished decoding pkiMessage");
190 				return new GetCertInitial(transId, senderNonce, messageData);
191 			} else if (messageType == MessageType.GET_CRL) {
192 				IssuerAndSerialNumber messageData = IssuerAndSerialNumber
193 						.getInstance(decoded);
194 				LOGGER.debug("Finished decoding pkiMessage");
195 				return new GetCrl(transId, senderNonce, messageData);
196 			} else {
197 				PKCS10CertificationRequest messageData;
198 				try {
199 					messageData = new PKCS10CertificationRequest(decoded);
200 				} catch (IOException e) {
201 					throw new MessageDecodingException(e);
202 				}
203 				LOGGER.debug("Finished decoding pkiMessage");
204 				return new PkcsReq(transId, senderNonce, messageData);
205 			}
206 		}
207 	}

209 	private void validate(CMSSignedData pkiMessage) {
210 		SignedData sd = SignedData.getInstance(pkiMessage.toASN1Structure()
211 				.getContent());
212 		LOGGER.debug("pkiMessage version: {}", sd.getVersion());
213 		LOGGER.debug("pkiMessage contentInfo contentType: {}", sd
214 				.getEncapContentInfo().getContentType());
215 	}

221 	private CMSEnvelopedData getEnvelopedData(Object bytes)
222 			throws MessageDecodingException {
223 		// We expect the byte array to be a sequence
224 		// ... and that sequence to be a ContentInfo (but might be the
225 		// EnvelopedData)
226 		try {
227 			return new CMSEnvelopedData((byte[]) bytes);
228 		} catch (CMSException e) {
229 			throw new MessageDecodingException(e);
230 		}
231 	}

233 	private Nonce toNonce(Attribute attr) {
234 		// Sometimes we don't get a sender nonce.
235 		if (attr == null) {
236 			return null;
237 		}
238 		final DEROctetString octets = (DEROctetString) attr.getAttrValues()
239 				.getObjectAt(0);
240 
241 		return new Nonce(octets.getOctets());
242 	}
