98      /**
99       * Sends the request to the SCEP server and processes the response..
100      * 
101      * @return the transaction state as returned by the SCEP server.
102      * @throws TransactionException
103      *             if any transaction-related error occurs.
104      */
105     @Override
106     public State send() throws TransactionException {
107 	CMSSignedData signedData;
108 	try {
109 	    signedData = encode(request);
110 	} catch (MessageEncodingException e) {
111 	    throw new TransactionException(e);
112 	}
113 	LOGGER.debug("Sending {}", signedData);
114 	PkiOperationResponseHandler handler = new PkiOperationResponseHandler();
115 	CMSSignedData res = send(handler, new PkiOperationRequest(signedData));
116 	LOGGER.debug("Received response {}", res);
117 
118 	CertRep response;
119 	try {
120 	    response = (CertRep) decode(res);
121 	} catch (MessageDecodingException e) {
122 	    throw new TransactionException(e);
123 	}
124 	validateExchange(request, response);
125 
126 	LOGGER.debug("Response: {}", response);
127 
128 	if (response.getPkiStatus() == PkiStatus.FAILURE) {
129 	    return failure(response.getFailInfo());
130 	} else if (response.getPkiStatus() == PkiStatus.SUCCESS) {
131 	    return success(extractCertStore(response));
132 	} else {
133 	    return pending();
134 	}
135     }

137     private void validateExchange(PkiMessage<?> req, CertRep res)
138 	    throws TransactionException {
139 	LOGGER.debug("Validating SCEP message exchange");
140 
141 	if (!res.getTransactionId().equals(req.getTransactionId())) {
142 	    throw new TransactionException("Transaction ID Mismatch");
143 	} else {
144 	    LOGGER.debug("Matched transaction IDs");
145 	}
146 
147 	// The requester SHOULD verify that the recipientNonce of the reply
148 	// matches the senderNonce it sent in the request.
149 	if (!res.getRecipientNonce().equals(req.getSenderNonce())) {
150 	    throw new InvalidNonceException(req.getSenderNonce(), res.getRecipientNonce());
151 	} else {
152 	    LOGGER.debug("Matched request senderNonce and response recipientNonce");
153 	}
154 
155 	if (res.getSenderNonce() == null) {
156 	    LOGGER.warn("Response senderNonce is null");
157 	    return;
158 	}
159 
160 	// http://tools.ietf.org/html/draft-nourse-scep-20#section-8.5
161 	// Check that the nonce has not been encountered before.
162 	if (QUEUE.contains(res.getSenderNonce())) {
163 	    throw new InvalidNonceException(res.getSenderNonce());
164 	} else {
165 	    QUEUE.add(res.getSenderNonce());
166 	    LOGGER.debug("{} has not been encountered before",
167 		    res.getSenderNonce());
168 	}
169 
170 	LOGGER.debug("SCEP message exchange validated successfully");
171     }
