22   * </pre>
23   */

30      /**
31       * The expected digest outcome.
32       */
33      private final byte[] expected;
34  
35      /**
36       * Creates a new instance with a digest algorithm, and the expected digest
37       * result.
38       * 
39       * @param digest
40       *            the digest algorithm to for verification.
41       * @param expected
42       *            the digest result
43       */
44      public MessageDigestCertificateVerifier(MessageDigest digest,
45  	    byte[] expected) {
46  	this.digest = digest;
47  	this.expected = ArrayUtils.clone(expected);
48      }
49  
50      /**
51       * {@inheritDoc}
52       */
53      public boolean verify(final X509Certificate cert) {
54  	byte[] actual;
55  	try {
56  	    digest.reset();
57  	    actual = digest.digest(cert.getTBSCertificate());
58  	} catch (CertificateEncodingException e) {
59  	    return false;
60  	}
