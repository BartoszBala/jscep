23  package org.jscep.transport.response;
24  
25  import org.bouncycastle.cms.CMSException;
26  import org.bouncycastle.cms.CMSSignedData;
27  
28  /**
29   * This class handles responses to <tt>PKCSReq</tt>, <tt>GetCertInitial</tt>,
30   * <tt>GetCert</tt> and <tt>GetCRL</tt> requests.
31   * 
32   * @author David Grant
33   */
34  public final class PkiOperationResponseHandler implements
35  	ScepResponseHandler<CMSSignedData> {
36      private static final String PKI_MESSAGE = "application/x-pki-message";
37  
38      /**
39       * {@inheritDoc}
40       */
41      public CMSSignedData getResponse(byte[] content, String mimeType)
42  	    throws ContentException {
43  	if (mimeType.startsWith(PKI_MESSAGE)) {
44  	    try {
45  		return new CMSSignedData(content);
46  	    } catch (CMSException e) {
47  		throw new InvalidContentException(e);
48  	    }
49  	} else {
50  	    throw new InvalidContentTypeException(mimeType, PKI_MESSAGE);
51  	}
52      }
53  }
