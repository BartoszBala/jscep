292 	private PKCS10CertificationRequest getCsr(X500Name subject,
293 			PublicKey pubKey, PrivateKey priKey, char[] password)
294 			throws GeneralSecurityException, IOException {
295 		SubjectPublicKeyInfo pkInfo = SubjectPublicKeyInfo.getInstance(pubKey
296 				.getEncoded());
297 
298 		JcaContentSignerBuilder signerBuilder = new JcaContentSignerBuilder(
299 				"SHA1withRSA");
300 		ContentSigner signer;
301 		try {
302 			signer = signerBuilder.build(priKey);
303 		} catch (OperatorCreationException e) {
304 			IOException ioe = new IOException();
305 			ioe.initCause(e);
306 
307 			throw ioe;
308 		}
309 
310 		PKCS10CertificationRequestBuilder builder = new PKCS10CertificationRequestBuilder(
311 				subject, pkInfo);
312 		builder.addAttribute(PKCSObjectIdentifiers.pkcs_9_at_challengePassword,
313 				new DERPrintableString(new String(password)));
314 
315 		return builder.build(signer);
316 	}
