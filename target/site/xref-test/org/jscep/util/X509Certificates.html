19  public final class X509Certificates {
20  	/**
21  	 * Creates a self-signed ephemeral certificate.
22  	 * <p/>
23  	 * The resulting certificate will have a not-before date of yesterday, and
24  	 * not-after date of tomorrow.
25  	 * 
26  	 * @param subject
27  	 *            the subject to certify.
28  	 * @param keyPair
29  	 *            the key pair to sign the certificate with.
30  	 * @return a new certificate.
31  	 * @throws GeneralSecurityException
32  	 *             if any security problem occurs.
33  	 */
34  	public static X509Certificate createEphemeral(final X500Principal subject,
35  			final KeyPair keyPair) throws GeneralSecurityException {
36  		final Calendar cal = Calendar.getInstance();
37  		cal.add(Calendar.DATE, -1);
38  		final Date notBefore = cal.getTime();
39  		cal.add(Calendar.DATE, 2);
40  		final Date notAfter = cal.getTime();
41  
42  		ContentSigner signer;
43  		try {
44  			signer = new JcaContentSignerBuilder(sigAlg(keyPair)).build(keyPair
45  					.getPrivate());
46  		} catch (OperatorCreationException e) {
47  			throw new GeneralSecurityException(e);
48  		}
49  		JcaX509v1CertificateBuilder builder = new JcaX509v1CertificateBuilder(
50  				subject, BigInteger.ONE, notBefore, notAfter, subject,
51  				keyPair.getPublic());
52  		X509CertificateHolder holder = builder.build(signer);
53  		return new JcaX509CertificateConverter().getCertificate(holder);
54  	}
55  
56  	private static String sigAlg(KeyPair keyPair) {
57  		return "SHA1with" + keyPair.getPrivate().getAlgorithm();
58  	}
